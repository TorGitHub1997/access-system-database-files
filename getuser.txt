create or replace function GetUser(cardid char(4))
returns table
(	
	firstname_ varchar,
	lastname_ varchar,
	email_address_ varchar,
	cardid_ char(4),
	cardpin_ char(4),
	validstart_ timestamp,
	validend_ timestamp
)as
$$
begin
	return query
	select first_name, last_name, email_address, card_id, card_pin, valid_start, 
		   valid_end from userdata where card_id = cardid order by card_ID;
	 
end
$$
language plpgsql;
