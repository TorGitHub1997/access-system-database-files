create or replace function GetUser(cardid_number char(4))
returns table
(	
	firstname varchar,
	lastname varchar,
	email varchar,
	cardid char,
	cardpin char,
	validstart timestamp,
	validend timestamp
)as
$$
begin
	return query
	select first_name, last_name, email_address, card_id, card_pin, valid_start, 
		   valid_end from userdata where cardid_number = card_id order by card_ID;
end
$$
language plpgsql;
