create or replace function DoorReport(doorNumber_ varchar(4), startDate_ Date, endDate_ Date)
returns table
(
	cardId char,
	approvedEntry bool,
	timeOfEntry timestamp,
	doorNumber varchar
)as
$$
begin
	return query
	select * from accesslog
		where door_Number = doorNumber_
		and time_of_entry between startDate_ and endDate_
		order by time_of_entry desc;
end
$$
language plpgsql;
